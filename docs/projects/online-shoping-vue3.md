```javascript
// 图片懒加载指令（正在学习VueUse库的应用）
import { useIntersectionObserver } from '@vueuse/core'

export const lazyPlugin = {
  install(app) {
    app.directive('img-lazy', {
      mounted(el, binding) {
        const { stop } = useIntersectionObserver(
          el,
          ([{ isIntersecting }]) => {
            if (isIntersecting) {
              el.src = binding.value
              stop() // 学习性能优化技巧
            }
          },
        )
      }
    })
  }
}
```
